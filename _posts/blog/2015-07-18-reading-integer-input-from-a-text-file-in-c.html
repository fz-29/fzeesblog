--- layout: post title: Reading integer input from a text file in C date: 2015-07-18 04:00:50.000000000 +05:30 type: post published: true status: publish categories: [] tags: [] --- Many times you need to read from a text file in order to take input. So here is one of the way : I have used the function fscanf(). {% highlight cpp linenos %} #include #include int main() { FILE *textfile=fopen("IntegerArray.txt","r"); int arr[100050],i,lim,num; if (textfile == NULL) { printf("Error Reading File\n"); exit (0); } i=0; while(fscanf(textfile, "%d", &num) >0 ) { arr[i++]=num; } lim=i; printf("\n Lim : %d \n",lim); for (i = 0; i < lim; i++) { printf("%d : %d\n",i, arr[i]); } fclose(textfile); return 0; } {% endhighlight %}
